time<-as.numeric(1:length(rainfall))
cosm<-matrix(nrow=length(rainfall),ncol=6)
sinm<-matrix(nrow=length(rainfall),ncol=5)
for(i in 1:5){
 cosm[,i]<-cos(2*pi*i*time/12)
 sinm[,i]<-sin(2*pi*i*time/12)

cosm[,6]<-cos(pi*time)
